class PlatformAwareKeywordSelector implements ISearchKeywordSelector {
  select(keywords: string[], platform: string): string {
    switch (platform) {
      case 'xiaohongshu':
        // 小时级切换
        const hour = Math.floor(Date.now() / 3600000);
        const index = hour % keywords.length;
        return keywords[index];
      case 'zhihu':
        // 优先专业问句
        const professionalKeywords = keywords.filter(
          k => k.includes('如何') || k.includes('评价') || k.includes('vs')
        );
        return professionalKeywords[0] || keywords.reduce((a, b) => a.length > b.length ? a : b);
      case 'autohome':
        // 短词优先（2-4 字）
        const shortKeywords = keywords.filter(k => k.length >= 2 && k.length <= 4);
        return shortKeywords[0] || keywords[0];
    }
  }
}
```

// 成功率调整
private async adjustBySuccessRate(priorities: Map<string, number>): Promise<Map<string, number>> {
  const adjusted = new Map<string, number>();
  for (const [platform, priority] of priorities.entries()) {
    const successRate = successRateMap.get(platform) || 100.0;
    let adjustedPriority = priority;
    if (successRate > 90) {
      adjustedPriority = Math.min(10, priority + 1);  // 奖励
    } else if (successRate < 50) {
      adjustedPriority = Math.max(1, priority - 2);  // 惩罚
    }
    adjusted.set(platform, adjustedPriority);
  }
  return adjusted;
}

// 权重随机选择（包含最近使用惩罚）
private weightedRandomSelect(platforms: ISearchPlatform[], priorities: Map<string, number>): ISearchPlatform {
  let totalWeight = 0;
  const weights: number[] = [];
  
  for (const platform of platforms) {
    let weight = priorities.get(platform) || 5;
    if (recentUses.has(platform)) {
      weight = Math.max(1, weight - 3);  // 最近使用惩罚
    }
    weights.push(weight);
    totalWeight += weight;
  }
  
  // 权重随机选择
  let random = Math.random() * totalWeight;
  for (let i = 0; i < platforms.length; i++) {
    random -= weights[i];
    if (random <= 0) {
      return platforms[i];
    }
  }
  return platforms[platforms.length - 1];
}
```

  platformFit: {             // 平台适配性

// 小红书图片选择
private async selectForXiaohongshu(images: MaterialRecord[], neededCount: number): Promise<MaterialRecord[]> {
  const scoredImages = await Promise.all(
    images.map(async (img) => {
      const qualityScore = await this.evaluateImageQuality(img);
      let xiaohongshuScore = qualityScore.totalScore;
      
      // 美观度加分
      xiaohongshuScore += qualityScore.aestheticScore * 0.3;
      
      // 检查是否是小红书偏好类型
      const pathLower = img.path.toLowerCase();
      const preferredKeywords = ['人像', '美食', '风景', '生活', '日常', '自拍'];
      const hasPreferredKeyword = preferredKeywords.some(k => pathLower.includes(k));
      if (hasPreferredKeyword) {
        xiaohongshuScore += 15;
      }
      
      // 避免过于专业的图片
      const professionalKeywords = ['参数', '配置', '数据', '图表', '对比'];
      const hasProfessionalKeyword = professionalKeywords.some(k => pathLower.includes(k));
      if (hasProfessionalKeyword) {
        xiaohongshuScore -= 20;
      }
      
      return { image: img, score: Math.max(0, xiaohongshuScore) };
    })
  );
  
  scoredImages.sort((a, b) => b.score - a.score);
  return scoredImages.slice(0, neededCount).map(s => s.image);
}

// 缓存包装器
async getAllPlatformConfigs(
  fallbackFetcher: () => Promise<PlatformConfig[]>
): Promise<PlatformConfig[]> {
  const cacheKey = formatKey(CACHE_KEYS.PLATFORM_CONFIG);
  
  // 尝试从 Redis 获取
  const cached = await getRedisClient().get(cacheKey);
  if (cached) {
    this.recordHit();
    return JSON.parse(cached) as PlatformConfig[];
  }
  
  // 缓存未命中，使用回退方法
  this.recordMiss();
  const configs = await fallbackFetcher();
  
  // 写入缓存（5 分钟过期）
  await this.setCache(cacheKey, configs, CACHE_TTL.PLATFORM_CONFIG);
  
  return configs;
}

// 配置更新时自动清除缓存
async invalidateOnUpdate(platform: string): Promise<void> {
  logger.info(`配置更新，清除平台 ${platform} 缓存`);
  await this.clearPlatformCache(platform);
}
